def bisection(f, a, b, tol=1e-12, max_iter=100):
    if f(a)*f(b) > 0:
        raise ValueError("f(a) and f(b) must have opposite signs")
    errors, roots = [], []
    for n in range(max_iter):
        c = (a + b) / 2
        roots.append(c)
        if abs(f(c)) < tol or (b - a)/2 < tol:
            break
        if f(a)*f(c) < 0:
            b = c
        else:
            a = c
        errors.append(abs(b - a))
    return np.array(roots), np.array(errors)

def newton(f, df, x0, tol=1e-12, max_iter=100):
    roots, errors = [x0], []
    for n in range(max_iter):
        x1 = x0 - f(x0)/df(x0)
        roots.append(x1)
        errors.append(abs(x1 - x0))
        if abs(x1 - x0) < tol:
            break
        x0 = x1
    return np.array(roots), np.array(errors)

In[5]:
def plot_errors(errors_bis, errors_newton, title):
    plt.figure(figsize=(7,4))
    plt.semilogy(errors_bis, label="Bisection")
    plt.semilogy(errors_newton, label="NewtonRaphson")
    plt.xlabel("Iteration n")
    plt.ylabel("|x  x| (log scale)")
    plt.title(title)
    plt.legend()
    plt.grid(True, which="both", ls="--")
    plt.show()
